import RPi.GPIO as GPIO;
import Adafruit_DHT
import firebase_admin
from firebase_admin import credentials, db
import time
sensor = Adafruit_DHT.DHT11
pin = 4
# Firebase setup
cred = credentials.Certificate("/home/pi/Downloads/dht11-e7273-firebase-adminsdk-fbsvc-dfea761128.json") # Replace with your key file
firebase_admin.initialize_app(cred, {
    'databaseURL': 'https://dht11-e7273-default-rtdb.firebaseio.com/' # Replace with your Firebase DB URL
})


# Reference to Firebase database location
ref = db.reference("environment")

try:
    while True:
        humidity, temperature = Adafruit_DHT.read_retry(sensor, pin)
#         humidity, temperature = Adafruit_DHT.read(DHT_SENSOR, DHT_PIN)

        if humidity is not None and temperature is not None:
            #             print(f"Temp={temperature:.1f}Â°C Humidity={humidity:.1f}%")
            print ("Humidity ="+str(humidity))
            print ("Temperature ="+str(temperature))

            # Push to Firebase
            ref.push({
                'temperature': temperature,
                'humidity': humidity,
                'timestamp': time.strftime('%Y-%m-%d %H:%M:%S')
            })
        else:
            print("Sensor failure. Check wiring.")

        time.sleep(10)

except KeyboardInterrupt:
    print("Terminated by user")
